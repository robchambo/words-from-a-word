# Phase 6 — Analytics & Tuning Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire Firebase (Analytics + Remote Config + Crashlytics) into the app, emit the 33-event taxonomy from `V1_1_CONTRACTS.md`, and move provisional numbers (interstitial cadence, bonus refill threshold, hint slot caps, bonus flat score, ad unit IDs, feature flags) from hardcoded constants to Remote Config — with safe fallbacks if Firebase is unreachable.

**Architecture:** `AnalyticsService` and `RemoteConfigService` are singletons (not providers), matching `AudioService`. They're initialized early in `main.dart`. Every `GameProvider` / `RewardsProvider` / monetisation event-emission site calls `AnalyticsService.instance.logEvent(...)` with snake_case names matching the canonical list. `RemoteConfigService` loads defaults from a Dart-side map, then `fetchAndActivate` replaces them. Existing hardcoded constants (`_bonusRefillThreshold`, `_bonusWordFlatScore`, interstitial `everyN: 3`, etc.) are read via getters so Remote Config can override without code changes.

**Tech Stack:** Dart 3.11, Flutter, `firebase_core ^3.3.0`, `firebase_analytics ^11.2.1`, `firebase_remote_config ^5.1.1`, `firebase_crashlytics ^4.0.4`.

```dart
import 'package:firebase_analytics/firebase_analytics.dart';
import 'package:flutter/foundation.dart';

/// Abstraction seam for tests. Real impl wraps FirebaseAnalytics;
/// [FakeAnalyticsBackend] records calls in-memory.
abstract class AnalyticsBackend {
  Future<void> logEvent(String name, Map<String, Object?>? params);
  Future<void> setUserProperty(String name, String? value);
}

class FirebaseAnalyticsBackend implements AnalyticsBackend {
  @override
  Future<void> logEvent(String name, Map<String, Object?>? params) async {
    await FirebaseAnalytics.instance.logEvent(
      name: name,
      parameters: params?.map((k, v) => MapEntry(k, v ?? '')),
    );
  }

  @override
  Future<void> setUserProperty(String name, String? value) async {
    await FirebaseAnalytics.instance.setUserProperty(name: name, value: value);
  }
}

class AnalyticsEvent {
  AnalyticsEvent(this.name, this.params);
  final String name;
  final Map<String, Object?>? params;
}

class FakeAnalyticsBackend implements AnalyticsBackend {
  final events = <AnalyticsEvent>[];
  final userProps = <String, String?>{};

  @override
  Future<void> logEvent(String name, Map<String, Object?>? params) async {
    events.add(AnalyticsEvent(name, params));
  }

  @override
  Future<void> setUserProperty(String name, String? value) async {
    userProps[name] = value;
  }
}

class AnalyticsService {
  AnalyticsService._(this._backend);
  factory AnalyticsService.forTesting({required AnalyticsBackend backend}) =>
      AnalyticsService._(backend);

  static AnalyticsService instance = AnalyticsService._(FirebaseAnalyticsBackend());

  final AnalyticsBackend _backend;

  Future<void> initialize() async {
    await FirebaseAnalytics.instance.setAnalyticsCollectionEnabled(true);
    await logEvent('app_open');
  }

  Future<void> logEvent(String name, {Map<String, Object?>? params}) async {
    try {
      await _backend.logEvent(name, params);
    } catch (e) {
      debugPrint('[AnalyticsService] logEvent error: $e');
    }
  }

  Future<void> setUserProperty(String name, String? value) async {
    try {
      await _backend.setUserProperty(name, value);
    } catch (e) {
      debugPrint('[AnalyticsService] setUserProperty error: $e');
    }
  }
}
```

```dart
import 'package:firebase_remote_config/firebase_remote_config.dart';
import 'package:flutter/foundation.dart';

class RemoteConfigService {
  RemoteConfigService._({Map<String, Object>? overrideDefaults})
      : _overrides = overrideDefaults;

  static RemoteConfigService instance = RemoteConfigService._();

  factory RemoteConfigService.forTestingWithDefaults() =>
      RemoteConfigService._(overrideDefaults: defaultsForTesting);

  final Map<String, Object>? _overrides;

  static const Map<String, Object> defaultsForTesting = {
    'interstitialLevelCadence': 3,
    'bonusRefillThreshold': 10,
    'freeHintSlotCapFree': 1,
    'freeHintSlotCapPremium': 3,
    'bonusWordFlatScore': 15,
    'rewardedAdsEnabled': true,
    'interstitialAdsEnabled': true,
    'admobInterstitialIdIos': '',
    'admobInterstitialIdAndroid': '',
    'admobRewardedIdIos': '',
    'admobRewardedIdAndroid': '',
  };

  Future<void> initialize() async {
    try {
      final rc = FirebaseRemoteConfig.instance;
      await rc.setConfigSettings(RemoteConfigSettings(
        fetchTimeout: const Duration(seconds: 10),
        minimumFetchInterval: const Duration(hours: 1),
      ));
      await rc.setDefaults(defaultsForTesting);
      await rc.fetchAndActivate();
    } catch (e) {
      debugPrint('[RemoteConfigService] fetch failed; using defaults: $e');
    }
  }

  T _read<T>(String key) {
    if (_overrides != null) return _overrides[key] as T;
    try {
      final rc = FirebaseRemoteConfig.instance;
      final v = defaultsForTesting[key];
      if (v is int) return rc.getInt(key) as T;
      if (v is double) return rc.getDouble(key) as T;
      if (v is bool) return rc.getBool(key) as T;
      if (v is String) return rc.getString(key) as T;
      return defaultsForTesting[key] as T;
    } catch (_) {
      return defaultsForTesting[key] as T;
    }
  }

  int get interstitialLevelCadence => _read<int>('interstitialLevelCadence');
  int get bonusRefillThreshold => _read<int>('bonusRefillThreshold');
  int get freeHintSlotCapFree => _read<int>('freeHintSlotCapFree');
  int get freeHintSlotCapPremium => _read<int>('freeHintSlotCapPremium');
  int get bonusWordFlatScore => _read<int>('bonusWordFlatScore');
  bool get rewardedAdsEnabled => _read<bool>('rewardedAdsEnabled');
  bool get interstitialAdsEnabled => _read<bool>('interstitialAdsEnabled');
  String get admobInterstitialIdIos => _read<String>('admobInterstitialIdIos');
  String get admobInterstitialIdAndroid => _read<String>('admobInterstitialIdAndroid');
  String get admobRewardedIdIos => _read<String>('admobRewardedIdIos');
  String get admobRewardedIdAndroid => _read<String>('admobRewardedIdAndroid');
}
```

```dart
import 'package:firebase_crashlytics/firebase_crashlytics.dart';
import 'package:flutter/foundation.dart';

class CrashlyticsService {
  CrashlyticsService._();
  static final CrashlyticsService instance = CrashlyticsService._();

  Future<void> initialize() async {
    if (kDebugMode) {
      await FirebaseCrashlytics.instance
          .setCrashlyticsCollectionEnabled(false);
      return;
    }
    await FirebaseCrashlytics.instance.setCrashlyticsCollectionEnabled(true);

    FlutterError.onError = (details) {
      FirebaseCrashlytics.instance.recordFlutterFatalError(details);
    };

    PlatformDispatcher.instance.onError = (error, stack) {
      FirebaseCrashlytics.instance.recordError(error, stack, fatal: true);
      return true;
    };
  }

  Future<void> recordError(Object error, StackTrace? stack,
      {bool fatal = false}) async {
    await FirebaseCrashlytics.instance.recordError(error, stack, fatal: fatal);
  }
}
```
